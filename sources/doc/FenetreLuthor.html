def initialize()
    super()
    self.window_position = Window::POS_CENTER
    self.set_title("Luthor Picross")
    self.signal_connect("destroy"){Gtk.main_quit()}
    @box = VBox.new()
    self.add(@box)

    @menu = MenuLuthor.new()
    @box.pack_start(@menu, false, true, 0)
end

def ecran=(ecran)
    @box.remove(@ecran) if @ecran != nil
    @ecran = ecran
    @box.add(@ecran)
    self.show_all()
    return self
end
